  <script>
    function detectAccount() {
      let followers = parseInt(document.getElementById("followers").value);
      let following = parseInt(document.getElementById("following").value);
      let posts = parseInt(document.getElementById("posts").value);
      let bioLength = parseInt(document.getElementById("bioLength").value);

      // Simple rule-based detection (replace with ML later)
      let score = 0;
      if (followers < 50 && following > 500) score++;
      if (posts < 5) score++;
      if (bioLength < 10) score++;
      if (followers/following < 0.1) score++;

      let result = document.getElementById("result");
      if (score >= 2) {
        result.innerHTML = "⚠️ This looks like a FAKE account!";
        result.style.color = "red";
      } else {
        result.innerHTML = "✅ This looks like a REAL account!";
        result.style.color = "green";
      }
    }
  </script>
